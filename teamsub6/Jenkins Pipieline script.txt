pipeline {
    agent any  

    triggers {
        pollSCM('*/3 * * * *')
    } 

    environment {
        DOCKER = credentials('dockerhub')
        DOCKER_USR = "${DOCKER_USR}"
        DOCKER_CREDS_PSW = "${DOCKER_PSW}"
    } 

    stages {
        stage('Prepare') {
            steps {
                git url: 'https://github.com/JJHDGU/2025-1-OSSPrac-usually-01.git',
                    branch: 'main',
                    credentialsId: 'github'
            }
        }

        stage('Build') {
            steps {
                dir("teamsub6") { 
                    sh "echo ${DOCKER_CREDS_PSW} | docker login -u ${DOCKER_USR} --password-stdin"
                    sh 'docker-compose build frontapp'
                    sh 'docker-compose build backapp'
                }
            }
        }

        stage('Tag') {
            steps {
                script {
                    sh "docker tag frontapp:latest ${DOCKER_USR}/frontapp:${BUILD_NUMBER}"
                    sh "docker tag backapp:latest ${DOCKER_USR}/backapp:${BUILD_NUMBER}"
                }
            }
        }

        stage('Push') {
            steps {
                script {
                    sh "echo $DOCKER_PSW | docker login -u $DOCKER_USR --password-stdin"
                    sh "docker push ${DOCKER_USR}/frontapp:${BUILD_NUMBER}"
                    sh "docker push ${DOCKER_USR}/backapp:${BUILD_NUMBER}"
                }
            }
        }
    }
}
